Reading data from external Excel file

Example (with additional node module)- https://www.grapecity.com/blogs/how-to-importexport-excel-files-using-javascript-and-spread-sheets

var excelIO = new GC.Spread.Excel.IO(); var excelUrl = $("#importUrl").val(); function ImportFile() { var excelUrl = "https://file-examples-com.github.io/uploads/2017/02/file_example_XLSX_10.xlsx"; var oReq = new XMLHttpRequest(); oReq.open('get', excelUrl, true); oReq.responseType = 'blob'; oReq.onload = function () { var blob = oReq.response; excelIO.open(blob, LoadSpread, function (message) { console.log(message); }); }; oReq.send(null); } function LoadSpread(json) { jsonData = json; workbook.fromJSON(json); workbook.setActiveSheet("Revenues (Sales)"); }
